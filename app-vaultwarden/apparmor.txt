#include <tunables/global>

profile vaultwarden flags=(attach_disconnected,mediate_deleted) {
  #include <abstractions/base>

  # Capabilities
  capability,
  file,
  signal (send) set=(kill,term,int,hup,cont),

  # S6-Overlay
  /init ix,
  /bin/** ix,
  /usr/bin/** ix,
  /run/{s6,s6-rc*,service}/** ix,
  /package/** ix,
  /command/** ix,
  /etc/services.d/** rwix,
  /etc/cont-init.d/** rwix,
  /etc/cont-finish.d/** rwix,
  /etc/s6-overlay/s6-rc.d/** r,
  /run/{,**} rwk,
  /dev/tty rw,

  # Bashio
  /usr/lib/bashio/** ix,
  /tmp/** rwk,

  # Access to app configuration
  /data/ r,
  /data/options.json rw,

  # Start new profile for Vaultwarden service
  /opt/vaultwarden cx -> vaultwarden_service,

  # Start new profile for Nginx service
  /usr/sbin/nginx cx -> nginx_service,

  profile vaultwarden_service flags=(complain,attach_disconnected,mediate_deleted) {
    #include <abstractions/base>

    # Receive signals from S6-Overlay
    signal (receive) peer=*_vaultwarden,

    # Capabilities
    capability net_bind_service,

    # Vaultwarden binary and web vault
    /opt/vaultwarden rm,
    /opt/web-vault/** r,

    # Network socket access
    network inet tcp,
    network inet udp,

    # Process access
    /proc/1/fd/1 w,
    /run/vaultwarden.pid rw,

    # File access
    /bin/bash rix,
    /bin/echo ix,
    /dev/null rw,
    /dev/tty rw,
    /etc/passwd r,
    /etc/ssl/openssl.cnf r,
    /etc/ssl/certs/ca-certificates.crt r,
    /dev/urandom r,
    /usr/bin/openssl rix,

    # System information
    /proc/*/cgroup r,
    /proc/*/mountinfo r,
    /proc/net/netlink r,
    /proc/sys/kernel/osrelease r,
    /proc/sys/net/core/somaxconn r,
    /proc/sys/net/ipv4/tcp_tw_reuse r,
    /sys/fs/cgroup/cpu.max r,

    # Work directory, attachments, configuration, SQLite database access, icon cache & temporary files
    /data/ r,
    /data/attachments/** rw,
    /data/config.json rw,
    /data/db.sqlite3 rwk,
    /data/db.sqlite3-shm rwk,
    /data/db.sqlite3-wal rwk,
    /data/icon_cache/** rw,
    /data/rsa_key.pem rw,
    /data/tmp/** rw,
  }

  profile nginx_service flags=(complain,attach_disconnected,mediate_deleted) {
    #include <abstractions/base>

    # Receive signals from S6-Overlay
    signal (receive) peer=*_vaultwarden,

    # Read process trace
    ptrace (read),

    # Capabilities
    capability dac_override,
    capability setgid,
    capability setuid,

    # Nginx binary and configuration
    /usr/sbin/nginx rm,
    /etc/nginx/** rw,

    # Network socket access
    network inet tcp,
    network inet udp,

    # Process access
    /proc/1/fd/1 w,
    /run/nginx.pid rw,

    # File access
    /bin/bash rix,
    /dev/null rw,
    /dev/tty rw,
    /etc/group r,
    /etc/passwd r,
    /etc/nsswitch.conf r,
    /etc/ssl/openssl.cnf r,
    /var/lib/nginx/body/ rw,
    /var/lib/nginx/fastcgi/ rw,
    /var/lib/nginx/proxy/ rw,
    /var/lib/nginx/scgi/ rw,
    /var/lib/nginx/uwsgi/ rw,

    # System information
    /proc/net/netlink r,
    /proc/sys/kernel/osrelease r,
    /proc/sys/net/core/somaxconn r,
    /proc/sys/net/ipv4/tcp_tw_reuse r,

    # Work directory, Temporary files
    /data/ r,
    /tmp/** rw,
  }
}
