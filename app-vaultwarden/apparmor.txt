#include <tunables/global>

profile vaultwarden flags=(attach_disconnected,mediate_deleted) {
  #include <abstractions/base>

  # Capabilities
  file,
  signal (send) set=(kill,term,int,hup,cont),

  # S6-Overlay
  /init ix,
  /bin/** ix,
  /usr/bin/** ix,
  /run/{s6,s6-rc*,service}/** ix,
  /package/** ix,
  /command/** ix,
  /etc/services.d/** rwix,
  /etc/cont-init.d/** rwix,
  /etc/cont-finish.d/** rwix,
  /etc/s6-overlay/s6-rc.d/** r,
  /run/{,**} rwk,
  /dev/tty rw,

  # Bashio
  /usr/lib/bashio/** ix,
  /tmp/** rwk,

  # Access to app configuration
  /data/options.json rw,

  # Start new profile for Vaultwarden service
  /opt/vaultwarden cx -> vaultwarden_service,

  # Start new profile for Nginx service
  /usr/sbin/nginx cx -> nginx_service,

  profile vaultwarden_service flags=(complain,attach_disconnected,mediate_deleted) {
    #include <abstractions/base>

    # Receive signals from S6-Overlay
    signal (receive) peer=*_vaultwarden,

    # Capabilities
    capability net_bind_service,

    # Vaultwarden binary and web vault
    /opt/vaultwarden r,
    /opt/web-vault/** r,

    # Network socket access
    network inet stream,
    network inet dgram,

    # Process access
    /proc/1/fd/1 w,
    /var/run/vaultwarden.pid rw,

    # System access
    /bin/bash rix,
    /bin/echo ix,
    /dev/null rw,
    /dev/tty rw,
    /etc/passwd r,
    /dev/urandom r,
    /usr/bin/openssl rix,

    # System information
    /proc/net/netlink r,
    /proc/sys/kernel/osrelease r,
    /proc/sys/net/core/somaxconn r,
    /proc/sys/net/ipv4/tcp_tw_reuse r,

    # Attachments, configuration, SQLite database access, icon cache & temporary files
    /data/attachments/** rw,
    /data/config.json rw,
    /data/db.sqlite3 rw,
    /data/db.sqlite3-shm rw,
    /data/db.sqlite3-wal rw,
    /data/icon_cache/** rw,
    /data/rsa_key.pem rw,
    /data/tmp/** rw,
  }

  profile nginx_service flags=(complain,attach_disconnected,mediate_deleted) {
    #include <abstractions/base>

    # Receive signals from S6-Overlay
    signal (receive) peer=*_vaultwarden,

    # Nginx binary and configuration
    /usr/sbin/nginx r,
    /etc/nginx/** r,

    # Network socket access
    network inet stream,
    network inet dgram,

    # Process access
    /proc/1/fd/1 w,
    /var/run/nginx.pid rw,

    # System access
    /bin/bash rix,
    /dev/null rw,
    /dev/tty rw,

    # System information
    /proc/net/netlink r,
    /proc/sys/kernel/osrelease r,
    /proc/sys/net/core/somaxconn r,
    /proc/sys/net/ipv4/tcp_tw_reuse r,

    # Temporary files
    /tmp/** rw,
  }
}
